sgm_create_fleet_test_effect = {
	space_owner = {
		save_event_target_as = sgm_ship_spawner_country
	}
	create_fleet = {
		name = random
		effect = {
			set_owner = event_target:sgm_ship_spawner_country
			set_location = prev
		}
	}
	last_created_fleet = {
		sgm_create_ship_test_effect = yes
	}
}

sgm_create_ship_test_effect = {
	event_target:global_event_country = {
		if = {
			limit = {
				NOT = {
					is_variable_set = sgm_spawn_ship_var_num_ships
				}
			}
			set_variable = {
				which = sgm_spawn_ship_var_num_ships
				value = 1
			}
		}
	}
	set_fleet_flag = NAME_Large_Crystal_Entity_Yellow
	sgm_create_ship_database_effect = yes
	clear_variable = sgm_spawn_ship_var_num_ships
	remove_fleet_flag = NAME_Large_Crystal_Entity_Yellow
}

sgm_create_ship_effect = {
	if = {
		limit = {
			is_scope_type = fleet
		}
		if = {
			# Spawn as new fleet
			limit = {
				has_fleet_flag = sgm_do_not_spawn_reinforce
			}
			create_fleet = {
				# name = random
				effect = {
					set_location = prev
					set_owner = prev.owner
					prev.owner = {
						save_event_target_as = event_target:sgm_ship_spawner_country
					}
					set_fleet_flag = $SHIP_DESIGN$
					sgm_create_ship_database_effect = yes
					remove_fleet_flag = $SHIP_DESIGN$
				}
			}
		}
		# Spawn as reinforcement for current fleet
		else = {
			owner = {
				save_event_target_as = event_target:sgm_ship_spawner_country
			}
			set_fleet_flag = $SHIP_DESIGN$
			sgm_create_ship_database_effect = yes
			remove_fleet_flag = $SHIP_DESIGN$
		}
	}
	else_if = {
		# Spawn as new fleet
		limit = {
			OR = {
				exists = event_target:target_country
			}
		}
		create_fleet = {
			# name = random
			effect = {
				set_location = prev
				set_owner = event_target:target_country
				owner = {
					save_event_target_as = event_target:sgm_ship_spawner_country
				}
				set_fleet_flag = $SHIP_DESIGN$
				sgm_create_ship_database_effect = yes
				remove_fleet_flag = $SHIP_DESIGN$
			}
		}
	}
}

# Supported Scopes: fleet starbase
# Parameters: DESIGN
sgm_create_ship_basic_effect = {
	event_target:global_event_country = {
		if = {
			limit = {
				NOT = {
					is_variable_set = sgm_spawn_ship_var_num_ships
				}
			}
			set_variable = {
				which = sgm_spawn_ship_var_num_ships
				value = 1
			}
		}
		if = {
			limit = {
				NOR = {
					has_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_yes
					has_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_no
				}
			}
			set_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_no
		}
	}
	if = {
		limit = {
			NOT = {
				is_variable_set = sgm_spawn_ship_var_num_ships
			}
		}
		set_variable = {
			which = sgm_spawn_ship_var_num_ships
			value = event_target:global_event_country.sgm_spawn_ship_var_num_ships
		}
	}
	if = {
		limit = {
			has_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_yes
			exists = event_target:target_country
			NOT = {
				has_fleet_flag = sgm_do_not_spawn_reinforce				# Let us not overdo it.
			}
		}
		while = {
			count = sgm_spawn_ship_var_num_ships
			create_ship = {
				# name = random
				prefix = no
				colonizer_species = owner
				design = $DESIGN$
				# random_existing_design = <ship size key>
				graphical_culture = event_target:target_country
				effect = {
					set_ship_flag = $DESIGN$
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_fleet_flag = sgm_do_not_spawn_reinforce
				}
			}
			clear_variable = sgm_spawn_ship_var_num_ships
		}
	}
	else = {
		while = {
			count = sgm_spawn_ship_var_num_ships
			create_ship = {
				# name = random
				prefix = no
				colonizer_species = owner
				design = $DESIGN$
				# random_existing_design = <ship size key>
				graphical_culture = $GRAPHICAL_CULTURE$
				effect = {
					set_ship_flag = $DESIGN$
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_fleet_flag = sgm_do_not_spawn_reinforce
				}
			}
			clear_variable = sgm_spawn_ship_var_num_ships
		}
	}
}

sgm_create_fleet_leader = {
	if = {
		limit = {
			NOT = {
				exists = event_target:sgm_create_fleet_leader_species
			}
		}
		if = {
			limit = {
				exists = owner_main_species
			}
			owner_main_species = {
				save_event_target_as = sgm_create_fleet_leader_species
			}
		}
		else = {
			random_galaxy_species = {
				limit = {
					owner = {
						OR = {
							is_country_type = default
							is_country_type = fallen_empire
							is_country_type = awakened_fallen_empire
						}
					}
				}
				save_event_target_as = sgm_create_fleet_leader_species
			}
		}
	}
	switch = {
		trigger = is_ship_class
		shipclass_military = {
			if = {
				limit = {
					NOT = {
						exists = leader
					}
				}
				owner = {
					create_leader = {
						class = admiral
						species = event_target:sgm_create_fleet_leader_species
						set_age = 30
						effect = {
							set_leader_flag = sgm_leader
							from = {
								assign_leader = prev
							}
						}
					}
				}
			}
			# else = {
			# 	log = "[this.GetName]: Admiral already exists."
			# }
		}
		shipclass_science_ship = {
			if = {
				limit = {
					NOT = {
						exists = leader
					}
				}
				owner = {
					create_leader = {
						class = scientist
						species = event_target:sgm_create_fleet_leader_species
						set_age = 30
						effect = {
							set_leader_flag = sgm_leader
							from = {
								assign_leader = prev
							}
						}
					}
				}
			}
			# else = {
			# 	log = "[this.GetName]: Scientist already exists."
			# }
		}
		shipclass_transport = {
			if = {
				limit = {
					NOT = {
						exists = leader
					}
				}
				set_timed_fleet_flag = {
					flag = sgm_create_fleet_leader
					days = 1
				}
				if = {
					limit = {
						owner = {
							any_owned_army = {
								fleet = {
									has_fleet_flag = sgm_create_fleet_leader
								}
								NOT = {
									exists = leader
								}
								NOT = {
									prev = {
										any_owned_army = {
											fleet = {
												has_fleet_flag = sgm_create_fleet_leader
											}
											NOT = {
												is_same_value = prevprev
											}
											exists = leader
										}
									}
								}
							}
						}
					}
					owner = {
						create_leader = {
							class = general
							species = event_target:sgm_create_fleet_leader_species
							set_age = 30
							effect = {
								set_leader_flag = sgm_leader
								from.owner = {
									random_owned_army = {
										limit = {
											NOT = {
												exists = leader
											}
											fleet = {
												has_fleet_flag = sgm_create_fleet_leader
											}
										}
										assign_leader = prevprev
									}
								}
							}
						}
					}
					remove_fleet_flag = sgm_create_fleet_leader
				}
				# else = {
				# 	log = "[this.GetName]: General already exists."
				# }
			}
			# else = {
			# 	log = "[this.GetName]: General already exists."
			# }
		}
	}
