sgm_spawn_system_var_jumps_plus_one = {
	potential = {
		is_gm = yes
	}
	allow = {
		always = yes
	}
	effect = {
		event_target:global_event_country = {
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = sgm_var_jumps_spawn_system
							value >= 5
						}
					}
				}
				change_variable = {
					which = sgm_var_jumps_spawn_system
					value = 1
				}
			}
		}
	}
}

sgm_spawn_system_hyperlane_switch = {
	potential = {
		is_gm = yes
	}
	allow = {
		always = yes
	}
	effect = {
		switch = {
			trigger = has_global_flag
			sgm_spawn_system_hyperlane_yes = {
				remove_global_flag = sgm_spawn_system_hyperlane_yes
				set_global_flag = sgm_spawn_system_hyperlane_no
			}
			sgm_spawn_system_hyperlane_no = {
				remove_global_flag = sgm_spawn_system_hyperlane_no
				set_global_flag = sgm_spawn_system_hyperlane_yes
			}
			default = {
				set_global_flag = sgm_spawn_system_hyperlane_yes
			}
		}
	}
}
