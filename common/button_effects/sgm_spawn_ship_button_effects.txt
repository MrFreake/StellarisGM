sgm_spawn_ship_var_num_ships_plus_one = {
	potential = {
		is_gm = yes
	}
	allow = {
		always = yes
	}
	effect = {
		event_target:global_event_country = {
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = sgm_spawn_ship_var_num_ships
							value >= 100
						}
					}
				}
				change_variable = {
					which = sgm_spawn_ship_var_num_ships
					value = 1
				}
			}
		}
	}
}

sgm_spawn_ship_var_num_ships_minus_one = {
	potential = {
		is_gm = yes
	}
	allow = {
		always = yes
	}
	effect = {
		event_target:global_event_country = {
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = sgm_spawn_ship_var_num_ships
							value <= 0
						}
					}
				}

					which = sgm_spawn_ship_var_num_ships
					value = -1
				}
			}
		}
	}
}

}

sgm_spawn_ship_override_to_owner_graphical_culture_switch = {
	potential = {
		is_gm = yes
	}
	allow = {
		always = yes
	}
	effect = {
		switch = {
			trigger = has_global_flag
			sgm_spawn_ship_override_to_owner_graphical_culture_yes = {
				remove_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_yes
				set_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_no
			}
			sgm_spawn_ship_override_to_owner_graphical_culture_no = {
				remove_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_no
				set_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_yes
			}
			default = {
				set_global_flag = sgm_spawn_ship_override_to_owner_graphical_culture_no
			}
		}
	}
}
